mijnGame.html — Basis Boilerplate 
<!DOCTYPE html>
<html lang="nl">
<head>
  <meta charset="UTF-8">
  <title>Mijn 3D Game</title>
  <style>
    body { margin: 0; background: black; }
    canvas { display: block; }
  </style>
</head>
<body>
  <!-- Three.js via CDN — geen installatie nodig! -->
  <script src="https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.min.js"></script>

  <script>
    // 1. Scene, camera en renderer aanmaken
    const scene    = new THREE.Scene();
    const camera   = new THREE.PerspectiveCamera(75, innerWidth / innerHeight, 0.1, 1000);
    const renderer = new THREE.WebGLRenderer({ antialias: true });

    renderer.setSize(innerWidth, innerHeight);
    document.body.appendChild(renderer.domElement);

    // 2. Licht toevoegen
    const licht = new THREE.DirectionalLight(0xffffff, 1);
    licht.position.set(5, 10, 5);
    scene.add(licht);
    scene.add(new THREE.AmbientLight(0xffffff, 0.3));

    // 3. Een kubus (speler-placeholder)
    const geometry = new THREE.BoxGeometry();
    const material = new THREE.MeshStandardMaterial({ color: 0x7c3aed });
    const kubus    = new THREE.Mesh(geometry, material);
    scene.add(kubus);

    // 4. Camera plaatsen
    camera.position.set(0, 2, 5);
    camera.lookAt(0, 0, 0);

    // 5. Game loop
    function animate() {
      requestAnimationFrame(animate);
      kubus.rotation.y += 0.01; // draai animatie
      renderer.render(scene, camera);
    }
    animate();

    // 6. Venster aanpassen bij resize
    window.addEventListener('resize', () => {
      camera.aspect = innerWidth / innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(innerWidth, innerHeight);
    });
  </script>
</body>
</html>

⌨️ WASD / Pijltjestoetsen beweging 
// Bijhouden welke toetsen ingedrukt zijn
const toetsen = {};
window.addEventListener('keydown', e => toetsen[e.key] = true);
window.addEventListener('keyup',   e => toetsen[e.key] = false);

// In je animate()-functie aanroepen:
function beweegSpeler() {
  const snelheid = 0.1;
  if (toetsen['w'] || toetsen['ArrowUp'])    speler.position.z -= snelheid;
  if (toetsen['s'] || toetsen['ArrowDown'])  speler.position.z += snelheid;
  if (toetsen['a'] || toetsen['ArrowLeft'])  speler.position.x -= snelheid;
  if (toetsen['d'] || toetsen['ArrowRight']) speler.position.x += snelheid;
}
📷 Camera die de speler volgt (third-person) 
// In je animate()-functie:
camera.position.x = speler.position.x;
camera.position.y = speler.position.y + 5;  // 5 omhoog
camera.position.z = speler.position.z + 8;  // 8 achter speler
camera.lookAt(speler.position);             // altijd naar speler kijken
💥 Simpele botsing detecteren (AABB) 
function botsingCheck(objectA, objectB) {
  // Maak een bounding box om elk object
  const boxA = new THREE.Box3().setFromObject(objectA);
  const boxB = new THREE.Box3().setFromObject(objectB);
  return boxA.intersectsBox(boxB); // geeft true als ze elkaar raken
}

// Gebruik in animate():
if (botsingCheck(speler, munt)) {
  scene.remove(munt); // munt verdwijnt
  score += 10;
}
⬆️ Springen met zwaartekracht 
let springSnelheid = 0;
let isOpGrond = true;

window.addEventListener('keydown', e => {
  if (e.key === ' ' && isOpGrond) {
    springSnelheid = 0.15; // opstoot kracht
    isOpGrond = false;
  }
});

// In animate():
function updateSprong() {
  springSnelheid -= 0.006;           // zwaartekracht
  speler.position.y += springSnelheid;
  if (speler.position.y <= 0) {      // grond bereikt
    speler.position.y = 0;
    isOpGrond = true;
    springSnelheid = 0;
  }
}

👾 Obstakels automatisch spawnen 
const obstakels = [];

function maakObstakel() {
  const geo = new THREE.BoxGeometry(1, 1, 1);
  const mat = new THREE.MeshStandardMaterial({ color: 0xe11d48 });
  const ob  = new THREE.Mesh(geo, mat);
  ob.position.set((Math.random() - 0.5) * 10, 0, -20); // willekeurige x
  scene.add(ob);
  obstakels.push(ob);
}

// Spawn elke 2 seconden
setInterval(maakObstakel, 2000);

// In animate() obstakels laten bewegen:
obstakels.forEach((ob, index) => {
  ob.position.z += 0.1; // richting speler
  if (ob.position.z > 10) {
    scene.remove(ob);
    obstakels.splice(index, 1); // opruimen
  }
});

⚡ Startup Challenges

Begin met challenge 1. Klaar? Door naar de volgende. Gebruik AI als je vastloopt!

  1. 1

    Zet Three.js op

    Maak een nieuw HTML-bestand, kopieer de boilerplate en open het in je browser. Je ziet een draaiende kubus? Dan zit je goed!

    ~10 min
  2. 2

    Beweeg je speler

    Voeg het WASD-snippet toe en laat de kubus bewegen. Pas de snelheid aan zodat het lekker aanvoelt.

    ~15 min
  3. 3

    Voeg een vloer toe

    Maak een plat vlak (PlaneGeometry) en draai het horizontaal (rotation.x = -Math.PI / 2). Geef het een kleur.

    ~10 min
  4. 4

    Camera volgen

    Gebruik het camera-snippet zodat de camera altijd achter je speler blijft. Speel met de offset-waarden.

    ~10 min
  5. 5

    Voeg je gameplay-idee toe

    Kies je game-idee en begin met bouwen. Gebruik de snippets en vraag AI om hulp bij specifieke features.

    ~45 min
  6. 🎉

    Presenteer je game!

    Laat de klas jouw game zien en spelen. Vertel welk idee je had, wat je moeilijk vond en wat je zou willen uitbreiden.

    ~10 min demo
